CREATE TABLE IF NOT EXISTS valores_climatologicos (
    valores_climatologicos_id INT AUTO_INCREMENT PRIMARY KEY,
    fecha DATE,
    altitud INT,
    tmed FLOAT,
    prec FLOAT,
    tmin FLOAT,
    horatmin TIME,
    tmax FLOAT,
    horatmax TIME,
    dir FLOAT,
    velemedia FLOAT,
    racha FLOAT,
    horaracha TIME,
    hrMedia FLOAT,
    hrMax FLOAT,
    horaHrMax TIME,
    hrMin FLOAT,
    horaHrMin TIME,
    indicativo_id INT,
    ciudad_id INT,
    provincia_id INT,
    FOREIGN KEY (indicativo_id) REFERENCES indicativos(indicativo_id),
    FOREIGN KEY (ciudad_id) REFERENCES ciudades(ciudad_id),
    FOREIGN KEY (provincia_id) REFERENCES provincias(provincia_id)
);
